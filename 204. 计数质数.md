
# [204. 计数质数](https://leetcode-cn.com/problems/count-primes/)

- 穷举

```
/**
 *
 * @complex T(n ^ (3 / 2)) / S(1)
 * @param {number} n
 * @return {*}  {number}
 */
function countPrimes(n: number): number {
    let count: number = 0;
    for (let i = 2; i < n; ++i) {
        count += +isPrime(i);
    }
    return count;
};

function isPrime(num: number): boolean {
    for (let i = 2; i * i <= num; ++i) {
        if (num % i === 0) {
            return false;
        }
    }
    return true;
}
```
