
# [204. 计数质数](https://leetcode-cn.com/problems/count-primes/)

- 穷举

```ts
/**
 *
 * @complex T(n ^ (3 / 2)) / S(1)
 * @param {number} n
 * @return {*}  {number}
 */
function countPrimes(n: number): number {
    let count: number = 0;
    for (let i = 2; i < n; ++i) {
        count += +isPrime(i);
    }
    return count;
};

function isPrime(num: number): boolean {
    for (let i = 2; i * i <= num; ++i) {
        if (num % i === 0) {
            return false;
        }
    }
    return true;
}
```

- 埃氏筛

```ts
/**
 *
 * @complex T(nloglogn)/S(n)
 * @param {number} n
 * @return {*}  {number}
 */
function countPrimes(n: number): number {
    const primes: number[] = new Array(n).fill(1);
    let count: number = 0;

    for (let i = 2; i < n; ++i) {
        if (primes[i]) {
            ++count;
            for (let j = i * i; j < n; j += i) {
                primes[j] = 0;
            }
        }
    }

    return count;
};
```
