# [5. 替换空格](https://leetcode.cn/problems/string-to-url-lcci/description/)

- 双指针

```ts
/**
 * @description 1. 先遍历统计空格数量，计算出替换后的总长度。
 *              2. 使用双指针从后往前遍历：
 *                 - p 指向原字符串末尾，q 指向新字符串末尾。
 *                 - 若遇到空格，q 指针依次填入 '0', '2', '%'；
 *                 - 若不是空格，直接复制字符。
 *              这种“倒序遍历”法避免了从前往后替换导致的大量字符移动。
 */
function replaceSpaces(S: string, length: number): string {
  const strArray = [...S];
  const n = strArray.reduce((sum, s, i) => {
    if (i < length && s === " ") {
      ++sum;
    }
    return sum;
  }, 0);
  let p = length - 1,
    q = length + n * 2 - 1;
  strArray.length = q + 1;
  while (p >= 0 && p < q) {
    if (strArray[p] === " ") {
      strArray[q--] = "0";
      strArray[q--] = "2";
      strArray[q--] = "%";
    } else {
      strArray[q--] = strArray[p];
    }
    p--;
  }

  return strArray.join("");
}
```

- 字符串

```typescript
/**
 * @description 利用原生字符串方法：
 *              1. 截取有效长度 `S.substr(0, length)`。
 *              2. 使用 `split(" ").join("%20")` 实现全量替换。
 */
function replaceSpaces(S: string, length: number): string {
  return S.substr(0, length).split(" ").join("%20");
}
```
