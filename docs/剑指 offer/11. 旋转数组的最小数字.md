# [11. 旋转数组的最小数字](https://leetcode.cn/problems/find-minimum-in-rotated-sorted-array/description/)

- 遍历

```typescript
function findMin(nums: number[]): number {
    let min = nums[0];
    for (let i = 1; i < nums.length; ++i) {
        min = Math.min(min, nums[i]);
    }
    return min;
};
```

- 二分查找

```typescript
/**
 * @description 二分查找：旋转数组中，左半部分 >= 右半部分（转折点除外）。
 *              比较 nums[mid] 与 nums[right]：
 *              1. nums[mid] < nums[right]：说明 mid 在右半部分，最小值在 [left, mid]，更新 right = mid
 *              2. nums[mid] > nums[right]：说明 mid 在左半部分，最小值在 [mid+1, right]，更新 left = mid + 1
 *              3. nums[mid] == nums[right]：无法判断区间，right-- 缩小范围
 *              最终 left == right 时指向最小值。
 */
function findMin(nums: number[]): number {
    let left = 0, right = nums.length - 1;
    while (left < right) {
        const mid = (left + right) >> 1;
        if (nums[mid] < nums[right]) {
            right = mid;
        } else if (nums[mid] > nums[right]) {
            left = mid + 1;
        } else {
            --right;
        }
    }

    return nums[left];
};
```
