# [10. 斐波那契数列](https://leetcode.cn/problems/fei-bo-na-qi-shu-lie-lcof/)

- 递归

```typescript
/**
 * @description 递归：直接利用斐波那契数列定义 F(n) = F(n-1) + F(n-2)。
 *              优点：代码简洁直观。
 *              缺点：存在大量重复计算，时间复杂度为 O(2^n)，空间复杂度为 O(n)（递归调用栈深度）。
 */
function fib(n: number): number {
    if (n < 0) {
        return 0;
    }
    if (n <= 1) {
        return n;
    }

    return (fib(n - 1) + fib(n - 2)) % 1000000007;
};
```

- 迭代

```typescript
/**
 * @description 迭代（滚动数组）：用两个变量滚动更新，避免重复计算。
 *              时间复杂度 O(n)，空间复杂度 O(1)。
 */
function fib(n: number): number {
    if (n < 0) {
        return 0;
    }
    if (n <= 1) {
        return n;
    }

    let f1 = 0, f2 = 1
    for (let i = 2; i <= n; ++i) {
        [f1, f2] = [f2, (f1 + f2) % 1000000007];
    }

    return f2;
};
```
