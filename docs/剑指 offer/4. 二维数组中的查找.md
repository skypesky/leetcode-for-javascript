# [4. 二维数组中的查找](https://leetcode.cn/problems/er-wei-shu-zu-zhong-de-cha-zhao-lcof/)

- 双指针

```typescript
/**
 *
 * @description 从右上角开始查找：
 *              1. 当前值 > target：由于每一列是递增的，target 必不在当前列，执行 col--。
 *              2. 当前值 < target：由于每一行是递增的，target 必不在当前行，执行 row++。
 *              3. 相等则返回 true。
 * @param {number[][]} plants
 * @param {number} target
 * @return {boolean}
 */
function findTargetIn2DPlants(plants: number[][], target: number): boolean {
  if (!plants.length) {
    return false;
  }
  const rows = plants.length,
    cols = plants[0].length;
  let row = 0,
    col = cols - 1;
  while (row < rows && col >= 0) {
    const current: number = plants[row][col];
    if (current === target) {
      return true;
    } else if (current < target) {
      row++;
    } else {
      col--;
    }
  }
  return false;
}
```
