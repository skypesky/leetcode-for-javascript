# [13. 机器人的运动范围](https://leetcode.cn/problems/ji-qi-ren-de-yun-dong-fan-wei-lcof/)

- 深度优先搜索 / 广度优先搜索

```typescript
/**
 * @description DFS 回溯：从起点 (0,0) 出发，深度优先搜索可达格子。
 *              由于机器人只能向右或向下移动，搜索时只需向右和向下两个方向。
 *              使用 visited 数组记录已访问格子，避免重复访问。
 *              递归终止条件：越界 / 已访问 / 数位和超过 cnt。
 *
 * 时间复杂度 O(m × n) 推导：
 * - 每个格子最多被访问一次（标记 visited 后不再进入）
 * - 递归深度最坏为 m + n
 * - digit() 复杂度忽略不计（数位和是常数操作）
 *
 * 空间复杂度 O(m × n) 推导：
 * - visited 数组大小为 m × n
 * - 递归栈深度最坏为 m + n（呈链条状）
 */
function wardrobeFinishing(m: number, n: number, cnt: number): number {
    const visited = new Array(m * n).fill(false);
    return count(m, n, 0, 0, cnt, visited);
}

function count(
    rows: number,
    cols: number,
    row: number,
    col: number,
    cnt: number,
    visited: boolean[],
): number {
    const pos = row * cols + col;
    if (row < 0 || row >= rows || col < 0 || col >= cols) {
        return 0;
    }
    if (visited[pos]) {
        return 0;
    }
    if (digit(row) + digit(col) > cnt) {
        return 0;
    }

    visited[pos] = true;

    const res = 1
        + count(rows, cols, row, col + 1, cnt, visited)
        + count(rows, cols, row + 1, col, cnt, visited);

    return res;
}

function digit(n: number): number {
    let sum = 0;
    while (n > 0) {
        sum += n % 10;
        n = Math.floor(n / 10);
    }
    return sum;
}
```
