# [6. 从尾到头打印链表](https://leetcode.cn/problems/cong-wei-dao-tou-da-yin-lian-biao-lcof/description/)

- 递归

```typescript
/**
 * Definition for singly-linked list.
 * class ListNode {
 *     val: number
 *     next: ListNode | null
 *     constructor(val?: number, next?: ListNode | null) {
 *         this.val = (val===undefined ? 0 : val)
 *         this.next = (next===undefined ? null : next)
 *     }
 * }
 */
function reverseBookList(head: ListNode | null): number[] {
  const stack = [];
  const reverse = (node: ListNode | null) => {
    if (!node) {
      return;
    }
    reverse(node.next);
    stack.push(node.val);
  };

  reverse(head);

  return stack;
}
```
