# [6. 从尾到头打印链表](https://leetcode.cn/problems/cong-wei-dao-tou-da-yin-lian-biao-lcof/description/)

- 递归

```typescript
/**
 * Definition for singly-linked list.
 * class ListNode {
 *     val: number
 *     next: ListNode | null
 *     constructor(val?: number, next?: ListNode | null) {
 *         this.val = (val===undefined ? 0 : val)
 *         this.next = (next===undefined ? null : next)
 *     }
 * }
 */
/**
 * @description 利用递归调用栈的特性：
 *              1. 递归访问 `node.next`，直到链表末尾。
 *              2. 在递归返回（回溯）的过程中，将当前节点的值存入数组。
 *              由于“回溯”是从最后一个节点开始的，存入数组的顺序自然就是从尾到头。
 */
function reverseBookList(head: ListNode | null): number[] {
  const stack = [];
  const reverse = (node: ListNode | null) => {
    if (!node) {
      return;
    }
    reverse(node.next);
    stack.push(node.val);
  };

  reverse(head);

  return stack;
}
```
