# [3. 数组中重复的数字](https://leetcode.cn/problems/find-the-duplicate-number/)

- Set

```typescript
function findDuplicate(nums: number[]): number {
  const set = new Set<number>();
  for (const num of nums) {
    if (set.has(num)) {
      return num;
    }
    set.add(num);
  }
}
```

- 快慢指针

```typescript
/**
 *
 * @description 龟兔算法: 快慢指针找到第一个相遇点，必定在环内。然后从环内任意一点和起点同时出发，再次相遇点就是环的入口，即重复的数字
 * @param {number[]} nums
 * @return {number}
 */
function findDuplicate(nums: number[]): number {
  let slow = nums[0],
    fast = nums[nums[0]];
  while (slow !== fast) {
    slow = nums[slow];
    fast = nums[nums[fast]];
  }
  let p1 = 0,
    p2 = slow;
  while (p1 != p2) {
    p1 = nums[p1];
    p2 = nums[p2];
  }

  return p1;
}
```
