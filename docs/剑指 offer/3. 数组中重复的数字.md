# [3. 数组中重复的数字](https://leetcode.cn/problems/find-the-duplicate-number/)

## Set

```typescript
/**
 * @description 遍历数组，利用哈希集合记录已出现的数字。
 *              如果当前数字已存在于集合中，说明找到了重复项。
 */
function findDuplicate(nums: number[]): number {
  const set = new Set<number>();
  for (const num of nums) {
    if (set.has(num)) {
      return num;
    }
    set.add(num);
  }
}
```

- 环检测

```typescript
/**
 * @description 将数组视为一个链表，`nums[i]` 指向下一个节点的索引。
 *              由于存在重复数字，链表中必然存在环。
 *              重复的数字即为环的入口。
 */
function findDuplicate(nums: number[]): number {
  const set = new Set<number>();
  for (const num of nums) {
    if (set.has(num)) {
      return num;
    }
    set.add(num);
  }
}
```

- 快慢指针

```typescript
/**
 *
 * @description 龟兔算法 (Floyd's Cycle-Finding Algorithm): 
 *              1. 快慢指针找到第一个相遇点，该点必定在环内。
 *              2. 然后从起点和相遇点同时出发，速度相同，再次相遇点就是环的入口（重复的数字）。
 * @param {number[]} nums
 * @return {number}
 */
function findDuplicate(nums: number[]): number {
  let slow = nums[0],
    fast = nums[nums[0]];
  while (slow !== fast) {
    slow = nums[slow];
    fast = nums[nums[fast]];
  }
  let p1 = 0,
    p2 = slow;
  while (p1 != p2) {
    p1 = nums[p1];
    p2 = nums[p2];
  }

  return p1;
}
```
