# [12. 矩阵中的路径](https://leetcode.cn/problems/word-search/)

- 回溯

```typescript
/**
 * @description 回溯：从矩阵中的每个格子出发，深度优先搜索目标字符串。
 *              利用路径只能经过一次的特点，通过标记 visited 避免重复访问。
 *
 * 时间复杂度 O(n × 3^k) 推导：
 * - n = rows × cols（矩阵单元格数，即起点数）
 * - k = word.length（单词长度，即递归深度）
 * - 每层递归最多 3 个分支（4 个方向排除来向）
 * - 总计 n × 3^k 次操作
 *
 * 空间复杂度 O(k) 推导：
 * - 递归栈深度最多为 k（等于单词长度）
 * - visited 数组为输入数据，不计入额外空间
 */
function exist(board: string[][], word: string): boolean {
    const rows = board.length;
    const cols = board[0].length;
    if (!rows || !cols) return false;

    const visited: boolean[] = new Array(rows * cols).fill(false);

    for (let row = 0; row < rows; ++row) {
        for (let col = 0; col < cols; ++col) {
            if (existsPath(board, visited, word, 0, row, col)) {
                return true;
            }
        }
    }

    return false;
};

function existsPath(
    board: string[][],
    visited: boolean[],
    word: string,
    index: number,
    row: number,
    col: number
) {
    if (index >= word.length) return true;
    if (row < 0 || row >= board.length) return false;
    if (col < 0 || col >= board[0].length) return false;

    const pos = row * board[0].length + col;

    if (visited[pos]) return false;
    if (board[row][col] !== word[index]) return false;

    visited[pos] = true;

    const found =
        existsPath(board, visited, word, index + 1, row - 1, col) ||
        existsPath(board, visited, word, index + 1, row, col + 1) ||
        existsPath(board, visited, word, index + 1, row + 1, col) ||
        existsPath(board, visited, word, index + 1, row, col - 1);

    visited[pos] = found;

    return found;
}
```
