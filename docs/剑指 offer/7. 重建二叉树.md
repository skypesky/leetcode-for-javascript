# [7. 重建二叉树](https://leetcode.cn/problems/construct-binary-tree-from-preorder-and-inorder-traversal/)

- 递归

```typescript
/**
 * Definition for a binary tree node.
 * class TreeNode {
 *     val: number
 *     left: TreeNode | null
 *     right: TreeNode | null
 *     constructor(val?: number, left?: TreeNode | null, right?: TreeNode | null) {
 *         this.val = (val===undefined ? 0 : val)
 *         this.left = (left===undefined ? null : left)
 *         this.right = (right===undefined ? null : right)
 *     }
 * }
 */

/**
 * @description 基于前序遍历和中序遍历的特性进行分治构造：
 *              1. 前序遍历的第一个节点始终是当前子树的“根节点”。
 *              2. 在中序遍历中找到该根节点，其左侧即为“左子树”，右侧为“右子树”。
 *              3. 计算出左子树的大小，从而在前序遍历序列中也划分出左右子树的范围。
 *              4. 递归地构造左子树和右子树。
 */
function buildTree(preorder: number[], inorder: number[]): TreeNode | null {
  return buildTreeWithIndex(
    preorder,
    0,
    preorder.length - 1,
    inorder,
    0,
    inorder.length - 1,
  );
}

function buildTreeWithIndex(
  preorder: number[],
  preorderStartIndex: number,
  preorderEndIndex: number,
  inorder: number[],
  inorderStartIndex: number,
  inorderEndIndex: number,
): TreeNode | null {
  if (
    preorderStartIndex > preorderEndIndex ||
    inorderStartIndex > inorderEndIndex
  ) {
    return null;
  }
  const root = new TreeNode(preorder[preorderStartIndex]);

  let size = 0;
  for (let i = inorderStartIndex; i <= inorderEndIndex; ++i) {
    if (inorder[i] === root.val) {
      size = i - inorderStartIndex;
      break;
    }
  }
  root.left = buildTreeWithIndex(
    preorder,
    preorderStartIndex + 1,
    preorderStartIndex + size,
    inorder,
    inorderStartIndex,
    inorderStartIndex + size - 1,
  );
  root.right = buildTreeWithIndex(
    preorder,
    preorderStartIndex + size + 1,
    preorderEndIndex,
    inorder,
    inorderStartIndex + size + 1,
    inorderEndIndex,
  );

  return root;
}
```
