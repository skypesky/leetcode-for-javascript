# [LCR 121. 寻找目标值 - 二维数组.md](https://leetcode.cn/problems/er-wei-shu-zu-zhong-de-cha-zhao-lcof/description/)

- 双指针

```typescript
function findTargetIn2DPlants(plants: number[][], target: number): boolean {
    if (!plants.length) {
        return false;
    }
    const rows = plants.length, cols = plants[0].length;
    let row = 0, col = cols - 1;
    while (row < rows && col >= 0) {
        const current: number = plants[row][col];
        if (current === target) {
            return true;
        } else if (current < target) {
            row++;
        } else {
            col--;
        }
    }
    return false;
};
```
