# 1. 题目<[345. 反转字符串中的元音字母](https://leetcode-cn.com/problems/reverse-vowels-of-a-string/)>

编写一个函数，以字符串作为输入，反转该字符串中的元音字母。

**示例 1：**

> 输入："hello"
>
> 输出："holle"

**示例 2：**

> 输入："leetcode"
>
> 输出："leotcede"

# 2. 思路与代码实现

```javascript
/**
 * @param {string} s
 * @return {string}
 */
var reverseVowels = function (s) {
    if (!s) {
        return "";
    }

    const array = [...s];
    let left = 0,
        right = array.length - 1;

    while (left < right) {
        
        // 从左往右查找元音字母
        while (left < right && !isVowel(array[left].toLowerCase())) {
            ++left;
        }

        // 从右往左查找元音字母
        while (left < right && !isVowel(array[right].toLowerCase())) {
            --right;
        }

        // 找过头了,说明无需再交换了,可能是两个方向都没有找到元音字母或者只有一个方向有元音字母,此时可以退出循环
        if (left >= right) {
            break;
        }

        // 找到了,需要反转这两个字母
        [
            array[left],
            array[right],
        ] = [
                array[right],
                array[left],
            ];
    
        // 初始化下一轮的查找
        ++left;
        --right;
    }

    return array.join('');
};

const VowelMap = {
    'a': true,
    'e': true,
    'i': true,
    'o': true,
    'u': true,
};
function isVowel(ch) {
    return !!VowelMap[ch];
}
```
