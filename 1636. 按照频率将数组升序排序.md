/**
 * @param {number[]} nums
 * @return {number[]}
 */
var frequencySort = function (nums) {
    const map = nums.reduce((map, num) => map.set(num, (map.get(num) || 0) + 1), new Map());
    return nums.sort((a, b) => map.get(a) === map.get(b) ? b - a : map.get(a) - map.get(b));
};
