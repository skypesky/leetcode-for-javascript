
# [690. 员工的重要性](https://leetcode-cn.com/problems/employee-importance/)

```ts
/**
 * Definition for Employee.
 * class Employee {
 *     id: number
 *     importance: number
 *     subordinates: number
 *     constructor(id: number, importance: number, subordinates: number[]) {
 *         this.id = (id === undefined) ? 0 : id;
 *         this.importance = (importance === undefined) ? 0 : importance;
 *         this.subordinates = (subordinates === undefined) ? [] : subordinates;
 *     }
 * }
 */

function GetImportance(employees: Employee[], id: number): number {

    employees.sort((a: Employee, b: Employee) => a.subordinates.length - b.subordinates.length);

    const employeeMap: Map<number, Employee> = new Map<number, Employee>();
    for (const employee of employees) {
        employeeMap.set(employee.id, employee);
    }

    const importanceMap: Map<number, number> = new Map<number, number>();

    for (const employee of employees) {
        fillImportance(employee.id, employeeMap, importanceMap);
    }

    return importanceMap.get(id);
};

function fillImportance(
    id: number,
    employeeMap: Map<number, Employee>,
    importanceMap: Map<number, number>
): number {
    if (importanceMap.has(id)) {
        return importanceMap.get(id);
    }
    const employee: Employee = employeeMap.get(id);
    let value: number = employee.importance;
    for (const subordinat of employee.subordinates) {
        value += fillImportance(subordinat, employeeMap, importanceMap);
    }
    importanceMap.set(id, value);
    return value;
}
```
