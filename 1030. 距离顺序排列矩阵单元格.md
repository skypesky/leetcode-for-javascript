/**
 * @param {number} R
 * @param {number} C
 * @param {number} r0
 * @param {number} c0
 * @return {number[][]}
 */
var allCellsDistOrder = function (R, C, r0, c0) {
    const mat = [];
    for (let row = 0; row < R; ++row) {
        for (let col = 0; col < C; ++col) {
            mat.push([row, col]);
        }
    }
    const instance = (m) => Math.abs(m[0] - r0) + Math.abs(m[1] - c0);
    return mat.sort((m1, m2) => instance(m1) - instance(m2));
};
