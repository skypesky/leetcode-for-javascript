
# [525. 连续数组](https://leetcode-cn.com/problems/contiguous-array/)

- 哈希表(超时了)

```ts
function findMaxLength(nums: number[]): number {

    let max: number = 0;

    let map: [number, number][] = new Array(nums.length).fill(null).map(value => [0, 0]);

    for (let i = 0; i < nums.length; ++i) {
        if (i) {
            map[i][nums[i]] = map[i - 1][nums[i]] + 1;
            map[i][1 - nums[i]] = map[i - 1][1 - nums[i]];
        } else {
            ++map[i][nums[i]];
        }
    }

    for (let i = 0; i < nums.length; ++i) {
        for (let j = i + 1; j < nums.length; ++j) {
            let zeroCount = i ? map[j][0] - map[i - 1][0] : map[j][0],
                oneCount: number = i ? map[j][1] - map[i - 1][1] : map[j][1];
            if (zeroCount === oneCount && zeroCount > 0) {
                max = Math.max(max, j - i + 1);
            }
        }
    }

    return max;
};
```
