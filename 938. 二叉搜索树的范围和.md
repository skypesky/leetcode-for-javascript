


# [938. 二叉搜索树的范围和](https://leetcode-cn.com/problems/range-sum-of-bst/)

- 递归

```typescript
/**
 * Definition for a binary tree node.
 * class TreeNode {
 *     val: number
 *     left: TreeNode | null
 *     right: TreeNode | null
 *     constructor(val?: number, left?: TreeNode | null, right?: TreeNode | null) {
 *         this.val = (val===undefined ? 0 : val)
 *         this.left = (left===undefined ? null : left)
 *         this.right = (right===undefined ? null : right)
 *     }
 * }
 */

function rangeSumBST(root: TreeNode | null, low: number, high: number): number {
    return mediumOrderTraversal(root, low, high)
};

function mediumOrderTraversal(root: TreeNode, low: number, high: number): number {

    if (!root) {
        return 0;
    }

    let value: number = 0;

    if (root.left) {
        value += mediumOrderTraversal(root.left, low, high);
    }
    if (low <= root.val && root.val <= high) {
        value += root.val;
    }
    if (root.right) {
        value += mediumOrderTraversal(root.right, low, high);
    }

    return value;
}
```
