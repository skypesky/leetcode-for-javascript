
# [55. 跳跃游戏](https://leetcode-cn.com/problems/jump-game/)

- 动态规划

```typescript
/**
 *
 * @tag 动态规划
 * @description 设dp[i]表示下标为i的位置是否可达,dp[i] = true 表示可达,dp[i] = false 表示不可达.那么dp[i]如何得到呢?
 *              展开的表达式就是:
 *              dp[i] =    (dp[i - 1] && nums[i - 1] + i - 1 >= i)
 *                      || (dp[i - 2] && nums[i - 2] + i - 2 >= i)
 *                      ...
 *                      || (dp[0] && dp[0] + 0 >= i)
 *              有了上面的动态规划转移方程,应该很容易就能求出正确的结果了.
 * @complex T(n ^ 2)/S(n)
 * @param {number[]} nums
 * @return {*}  {boolean}
 */
function canJump(nums: number[]): boolean {

    const { length } = nums,
        dp: boolean[] = new Array<boolean>(nums.length).fill(false);

    dp[0] = true;

    for (let i = 1; i < length; ++i) {
        for (let j = i - 1; j >= 0; --j) {
            if (dp[j] && j + nums[j] >= i) {
                dp[i] = true;
                break;
            }
        }
    }

    return dp[length - 1];
};
```

- 贪心算法

```typescript
/**
 *
 * @tag 动态规划, 贪心算法
 * @complex T(n)/S(1)
 * @param {number[]} nums
 * @return {*}  {boolean}
 */
function canJump(nums: number[]): boolean {

    // 我想知道我现在最远可以跳到哪呢?
    let max: number = nums[0];

    for (let i = 0; i < nums.length; ++i) {
        if (i > max) { // 跳到最远的位置也到不了i这个位置,无法继续了
            return false;
        }
        // 统计一下最远可以跳到哪个位置呢?
        max = Math.max(max, i + nums[i]);
        if (max >= nums.length - 1) { // 可以跳到最后一个位置了,表示可达,后面不需要再执行了
            return true;
        }
    }

    return false;
};
```
